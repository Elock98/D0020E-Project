
'function createorg'$org_id'() {' #VARIABLE#
  infoln "Enrolling the CA admin"
  'mkdir -p organizations/peerOrganizations/org'$org_id'.example.com/' #VARIABLE#

  'export FABRIC_CA_CLIENT_HOME=${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/' #VARIABLE#

  set -x
  'fabric-ca-client enroll -u https://admin:adminpw@localhost:'$portnum' --caname ca-org'$org_id' --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  echo 'NodeOUs:
  Enable: true
  ClientOUIdentifier:
    Certificate: cacerts/localhost-$portnum-ca-org$org_id.pem #VARIABLE#
    OrganizationalUnitIdentifier: client
  PeerOUIdentifier:
    Certificate: cacerts/localhost-$portnum-ca-org$org_id.pem #VARIABLE#
    OrganizationalUnitIdentifier: peer
  AdminOUIdentifier:
    Certificate: cacerts/localhost-$portnum-ca-org$org_id.pem #VARIABLE#
    OrganizationalUnitIdentifier: admin
  OrdererOUIdentifier:
    Certificate: cacerts/localhost-$portnum-ca-org$org_id.pem #VARIABLE#
    $'OrganizationalUnitIdentifier: orderer\' > "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/config.yaml"' #VARIABLE#

  # Since the CA serves as both the organization CA and TLS CA, copy the org's root cert that was generated by CA startup into the org level ca and tlsca directories

  # Copy org$org_id's CA cert to org$org_id's /msp/tlscacerts directory (for use in the channel MSP definition)
  'mkdir -p "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/tlscacerts"' #VARIABLE#
  'cp "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/tlscacerts/ca.crt"' #VARIABLE#

  # Copy org$org_id's CA cert to org$org_id's /tlsca directory (for use by clients)
  'mkdir -p "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/tlsca"' #VARIABLE#
  'cp "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/tlsca/tlsca.org'$org_id'.example.com-cert.pem"' #VARIABLE#

  # Copy org$org_id's CA cert to org$org_id's /ca directory (for use by clients)
  'mkdir -p "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/ca"' #VARIABLE#
  'cp "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/ca/ca.org'$org_id'.example.com-cert.pem"' #VARIABLE#

  infoln '"Registering peer'$peer_id'"' #VARIABLE#
  set -x
  'fabric-ca-client register --caname ca-org'$org_id' --id.name peer'$peer_id' --id.secret peer'$peer_id'pw --id.type peer --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  infoln "Registering user"
  set -x
  'fabric-ca-client register --caname ca-org'$org_id' --id.name user1 --id.secret user1pw --id.type client --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  infoln "Registering the org admin"
  set -x
  'fabric-ca-client register --caname ca-org'$org_id' --id.name org'$org_id'admin --id.secret org'$org_id'adminpw --id.type admin --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  infoln '"Generating the peer'$peer_id' msp"' #VARIABLE#
  set -x
  'fabric-ca-client enroll -u https://peer'$peer_id':peer'$peer_id'pw@localhost:'$portnum' --caname ca-org'$org_id' -M "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/msp" --csr.hosts peer'$peer_id'.org'$org_id'.example.com --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/config.yaml" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/msp/config.yaml"' #VARIABLE#

  infoln '"Generating the peer'$peer_id'-tls certificates"' #VARIABLE#
  set -x
  'fabric-ca-client enroll -u https://peer'$peer_id':peer'$peer_id'pw@localhost:'$portnum' --caname ca-org'$org_id' -M "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls" --enrollment.profile tls --csr.hosts peer'$peer_id'.org'$org_id'.example.com --csr.hosts localhost --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  # Copy the tls CA cert, server cert, server keystore to well known file names in the peer's tls directory that are referenced by peer startup config
  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/tlscacerts/"* "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/ca.crt"' #VARIABLE#
  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/signcerts/"* "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/server.crt"' #VARIABLE#
  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/keystore/"* "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/peers/peer'$peer_id'.org'$org_id'.example.com/tls/server.key"' #VARIABLE#

  infoln "Generating the user msp"
  set -x
  'fabric-ca-client enroll -u https://user1:user1pw@localhost:'$portnum' --caname ca-org'$org_id' -M "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/users/User1@org'$org_id'.example.com/msp" --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/config.yaml" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/users/User1@org'$org_id'.example.com/msp/config.yaml"' #VARIABLE#

  infoln "Generating the org admin msp"
  set -x
  'fabric-ca-client enroll -u https://org'$org_id'admin:org'$org_id'adminpw@localhost:'$portnum' --caname ca-org'$org_id' -M "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/users/Admin@org'$org_id'.example.com/msp" --tls.certfiles "${PWD}/organizations/fabric-ca/org'$org_id'/ca-cert.pem"' #VARIABLE#
  { set +x; } 2>/dev/null

  'cp "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/msp/config.yaml" "${PWD}/organizations/peerOrganizations/org'$org_id'.example.com/users/Admin@org'$org_id'.example.com/msp/config.yaml"' #VARIABLE#
}
